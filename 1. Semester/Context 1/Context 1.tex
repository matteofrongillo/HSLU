\section{Role Descriptions}
\begin{center}
    \begin{tabular}{|>{\columncolor[gray]{0.8}}l|>{\columncolor[gray]{0.9}}l|}
        \hline
        \rowcolor[gray]{0.7} 
        \textbf{Role} & \textbf{Task/Responsibility} \\ \hline
        \textbf{Project Leader} &
            \begin{tabular}[c]{@{}l@{}}
                - Responsible for the supervision and success of the project;\\
                - Coordinates the team and ensures communication between roles;\\
                - Resolves conflicts and addresses any issues that arise;\\
                - Monitors overall progress and ensures project goals are met.
            \end{tabular} \\ \hline
        \textbf{Project Planner} &
            \begin{tabular}[c]{@{}l@{}}
                - Develops a detailed project timeline and plan;\\
                - Allocates resources and assigns tasks to team members;\\
                - Tracks deadlines and milestones to ensure timely completion;\\
                - Adjusts the project plan as needed based on progress and feedback.
            \end{tabular} \\ \hline
        \textbf{Materials Expert} &
            \begin{tabular}[c]{@{}l@{}}
                - Identifies and selects appropriate materials for the project;\\
                - Ensures the quality of materials meets technical requirements;\\
                - Works with suppliers to manage material procurement;\\
                - Stays updated on new technologies relevant to the project.
            \end{tabular} \\ \hline
        \textbf{Marketing/Operations Manager} &
            \begin{tabular}[c]{@{}l@{}}
                - Creates and implements marketing strategies for project promotion;\\
                - Manages project branding and external communications.
            \end{tabular} \\ \hline
        \textbf{Technical Writer} &
            \begin{tabular}[c]{@{}l@{}}
                - Produces and edits technical documents, including manuals and reports;\\
                - Ensures accuracy and clarity in all project-related documentation;\\
                - Reviews documentation for consistency with project goals.
            \end{tabular} \\ \hline
    \end{tabular}
\end{center}
